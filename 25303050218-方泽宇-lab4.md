```c
#include "cachelab.h"
#include <stdio.h>
#include <stdlib.h>// 标准库，包含输入输出和内存分配函数
#include <string.h>// 字符串处理库
#include <unistd.h>// 解析命令行参数的库
#include <getopt.h>// 解析命令行参数的库
#include <limits.h>// 定义cache line结构体，包含有效位、标签位和LRU时间戳

//高速缓存结构
typedef struct Cache_line{
    int valid; // 有效位
    int tag; // 标签位
    int lru; // 时间戳(LRU)
} Cache_line;

typedef struct Cache{
    int s; // set
    int l; // line
    int b; // block
    Cache_line **line;
     // 二维数组，行数为S，列数为E, 每一个组员就是一个cache line
} Cache;

//全局变量定义
int hit_count = 0, miss_count = 0, eviction_count = 0; // 命中、未命中、替换次数初始化
Cache *cache = NULL; // 定义一个全局的cache指针，方便在各个函数中访问和修改高速缓存结构
int verbose = 0; // 是否打印详细信息的标志，默认为0（不打印）
char t[1000]; // 存储trace文件名的全局变量

//初始化高速缓存
void initCache(int s, int l, int b) {
    int S = 1 << s; // 组数2^s

    // 分配内存并初始化高速缓存
    cache = (Cache *)malloc(sizeof(Cache));
    cache->s = s;
    cache->l = l;
    cache->b = b;
    cache->line = (Cache_line **)malloc(sizeof(Cache_line *) * S);
    // 分配行指针数组,S行*单行内存，所以是S*sizeof(Cache_line*)


    for (int i = 0; i < S; i++) {
        cache->line[i] = (Cache_line *)malloc(sizeof(Cache_line) * l); 
        // 分配每行的列数组, l列*单个cache line的内存，所以是l*sizeof(Cache_line)

        for (int j = 0; j < l; j++) {
            cache->line[i][j].valid = 0; // 初始时，高速缓存是空的
            cache->line[i][j].tag = -1;
            cache->line[i][j].lru = 0;  // 初始时，LRU时间戳为0
        }
    }
    //先基于行内存分配整体内存，再在循环中基于单个cache line内存分配行内存，
    //衔接外循环逐行分配，内循环逐列分配
}

//释放高速缓存内存
void freeCache(Cache *cache) {
    int S = 1 << cache->s; // 组数

    for (int i = 0; i < S; i++) {
        free(cache->line[i]); // 释放每行的列数组
    }
    free(cache->line); // 释放行指针数组
    free(cache); // 释放高速缓存结构体
}

// 获取访问地址的行索引
int getIndex(int op_s, int op_tag)
{
    for (int i = 0; i < cache->l; i++){
        if (cache->line[op_s][i].valid && cache->line[op_s][i].tag == op_tag)
        // 如果该行有效且标签匹配，返回该行索引
            return i;
    }

    return -1;
}

// 判断op_s组是否满了，如果满了返回-1，否则返回第一个无效行的索引
int isFull(int op_s){
    for (int i = 0; i < cache->l; i++){
        if (cache->line[op_s][i].valid == 0)
         // 如果该行无效，说明该组未满，返回该行索引
            return i;
    }

    return -1; // 如果所有行都有效，说明该组已满，返回-1
}

// 找到op_s组中LRU的行索引,如果miss了且该组满了，就需要替换LRU行
int findLRU(int op_s){
    int max_index = 0;// 初始化最大索引为0
    int max_lru = 0;// 初始化最大时间戳为0

    for(int i = 0; i < cache->l; i++){
        if(cache->line[op_s][i].lru > max_lru){
            // 如果当前行的LRU时间戳大于最大时间戳，更新最大时间戳和最大索引
            max_lru = cache->line[op_s][i].lru;
            max_index = i;
        }
    }

    return max_index;// 返回LRU行的索引
}

//更新cache line的有效位、标签和LRU时间戳
void updateCacheline(int i, int op_s, int op_tag){
    // 更新访问的行
    cache->line[op_s][i].valid = 1; // 将该行设置为有效
    cache->line[op_s][i].tag = op_tag; // 更新该行的标签
    cache->line[op_s][i].lru = 0; // 将当前访问的行的LRU时间戳重置为0，表示最近使用过

    // 更新该组中其他行的LRU时间戳
    for(int k = 0; k < cache->l; k++) // 更新该组中所有行的LRU时间戳
        if(cache->line[op_s][k].valid == 1) // 只有有效行才更新LRU时间戳
            cache->line[op_s][k].lru++; // 所有有效行的LRU时间戳加1
}

// 访问数据，更新命中、未命中和替换次数，并根据verbose标志打印相应信息
void accessData(int op_tag, int op_s, int verbose)
{
    int index = getIndex(op_s, op_tag); // 获取访问地址的行索引

    if (index == -1){// 如果未命中
        miss_count++; // 未命中次数加1
        if (verbose) // 如果verbose标志为真，打印"miss"
            printf("miss ");

        int i = isFull(op_s); // 判断该组是否满了
        if(i == -1){ // 如果满了，需要替换LRU行
            eviction_count++; // 替换次数加1
            if(verbose) printf("eviction "); // 如果verbose标志为真，打印"eviction"
            i = findLRU(op_s); // 找到LRU行的索引
        }

        updateCacheline(i, op_s, op_tag); // 更新cache line的有效位、标签和LRU时间戳
    }
    else {// 如果命中
        hit_count++; // 命中次数加1
        if(verbose) // 如果verbose标志为真，打印"hit"
            printf("hit ");

        updateCacheline(index, op_s, op_tag); 
        // 更新cache line的有效位、标签和LRU时间戳
    }
}

void getTrace(int s, int l, int b){
   FILE *fp = fopen(t, "r"); // 打开trace文件

    if (fp == NULL) {// 如果文件打开失败，打印错误信息并退出
        fprintf(stderr, "Error opening file: %s\n", t);
        exit(EXIT_FAILURE);
    }

    char operation;// 操作类型
    unsigned long long address;// 访问地址
    int size;// 访问大小

    while (fscanf(fp, " %c %llx,%d", &operation, &address, &size) == 3) {
      // 读取每行的操作类型、地址和大小
        if (operation == 'I') // 如果是指令访问，跳过
            continue;

        int op_s = (address >> b) & ((1 << s) - 1); // 获取组索引
        int op_tag = address >> (s + b); // 获取标签

        accessData(op_tag, op_s, verbose);
        // 访问数据，更新命中、未命中和替换次数，并根据verbose标志打印相应信息

        if (operation == 'M')
        // 如果是修改操作，修改操作会导致一次命中，所以需要再访问一次数据
            accessData(op_tag, op_s, verbose);
    }

    fclose(fp); // 关闭文件
}

// 打印使用说明
void printUsage() {
    printf("Usage: ./csim [-hv] -s <num> -E <num> -b <num> -t <file>\n");
    printf("Options:\n");
    printf("  -h         Print this help message.\n");
    printf("  -v         Optional verbose flag.\n");
    printf("  -s <num>   Number of set index bits.\n");
    printf("  -E <num>   Number of lines per set.\n");
    printf("  -b <num>   Number of block offset bits.\n");
    printf("  -t <file>  Trace file.\n\n");
    printf("Examples:\n");
    printf("linux>  ./csim -s 4 -E 1 -b 4 -t traces/yi.trace\n");
    printf("linux>  ./csim -v -s 8 -E 2 -b 4 -t traces/yi.trace\n");
}

// 主函数，解析命令行参数，初始化高速缓存，读取trace文件并模拟缓存访问，最后打印结果总结
int main(int argc, char *argv[]) {
    char opt;// 用于存储命令行选项
    int s, l, b;// 用于存储命令行参数s、E、b

    while (-1 != (opt = getopt(argc, argv, "hvs:E:b:t:")))
    // 解析命令行参数，使用getopt函数处理选项和参数
    {
        switch (opt)
        {
        case 'h':// 如果选项是-h，打印使用说明并退出
            printUsage();
            exit(0);
        case 'v':// 如果选项是-v，设置verbose标志为1，表示需要打印详细信息
            verbose = 1;
            break;
        case 's':// 如果选项是-s，将s参数转换为整数并存储在变量s中
            s = atoi(optarg);
            break;
        case 'E':// 如果选项是-E，将E参数转换为整数并存储在变量l中
            l = atoi(optarg);
            break;
        case 'b':// 如果选项是-b，将b参数转换为整数并存储在变量b中
            b = atoi(optarg);
            break;
        case 't':// 如果选项是-t，将t参数（文件名）复制到全局变量t中
            strcpy(t, optarg);
            break;
        default:
            printUsage();
            exit(-1);
        }
    }

    initCache(s, l, b); //初始化一个cache
    getTrace(s, l, b);//读取trace文件并模拟缓存访问
    freeCache(cache);//释放cache内存

    printSummary(hit_count, miss_count, eviction_count);
    return 0;
}
```

经查阅CMU资料此处引出分块优化(blocking)：
1.Blocking/Tiling​ = 把大矩阵切成小块（sub-matrix / tile）分别做转置
2.块大小要依据 Cache Line 大小、Cache 容量、矩阵尺寸​ 来选
3.目的是提高时间局部性（Temporal Locality），让一次 load 进 cache 的数据被充分用完再被 evict

为什么选 8×8：
1.一个 Cache Line 存 8 个 int → 行内空间局部性好
2.8 行 × 32B = 256B，A的一个 8×8 块 + B的一个 8×8 块共 512B，对于非对角线块可以同时留在 1KB Cache 中
3.减少 A↔B之间的 cache 行冲突驱逐
4x4 cacheline只用一半，利用率低
16x16 >1kB 绝对evict
*对角线块不可完全避免的额外conflict miss导致了1184 > 1152

```c
void transpose_submit(int M, int N, int A[N][M], int B[M][N])

    for (int i = 0; i < N; i += 8)

            for (int k = 0; k < 8; k++)
                for (int s = 0; s < 8; s++)
                    B[j + s][i + k] = A[i + k][j + s];
```

#### 64x64
优先考虑4x4而非8x8
reason:
miss rate = 行级对齐 + 共存时间 + 整行粒度
4x4相比于8x8具备相同的逐行对齐，但压缩了整个周期的共存时间，也削减了整行粒度
所以降低了miss
result:m > 1300 NO!

查阅资料有优化路线：
8×8 blocking + diagonal handling + manual cache line splitting
reason:
1.8×8 是最大 tile 尺寸，使 A/B 非对角线 tile 利用率最高(4x4 cacheline只用一半利用率低)
2️.8×8 对角线 tile 在 1KB 直接映射 Cache 中 A/B 行级完全对齐 → 必然互踢
3️.把一个 8×8 对角线 tile 拆成 4 个 4×4 子块，可以手动控制访问顺序，减少对齐行共存时间
4️.4×4 是小于“行冲突粒度(8 lines)”的最大安全子块 ⇒ 最小额外开销 + 破对齐冲突
相当于是将 8x8 和 4x4 的好处结合，扬长避短

step:
1.8x8内部拆分为4x4
2.把 A 的左上 4×4 拷贝到 B 的左上
3.把 A 的右上 4×4 拷贝到 B 的右上
4.手动交换右上 ↔ 左下
5.最后处理右下 4×4

## 遇到的问题及解决办法

1. 遇到问题：
```c
//全局变量
Cache* cache = NULL;
void initCache(int s, int l, int b) {
    int S = 1 << s;

    // ❌ 这里定义了一个“局部变量 cache”
    Cache *cache = (Cache *)malloc(sizeof(Cache));
    ...
}
```
函数结束后，局部变量销毁，全局变量cache任然是NULL，致使访问
空指针，未定义行为，统计结果全错

   解决方案：
```c
//全局变量
Cache* cache = NULL;
void initCache(int s, int l, int b) {
    int S = 1 << s;

    // 用全局变量
    cache = (Cache *)malloc(sizeof(Cache));
    ...
}
```

2. 遇到问题：
getTrace()中有
```c
while (fscanf(fp, " %c %llx,%d", &operation, &address, &size) == 3) {
    if (operation == 'I') continue;

    int op_s = (address >> b) & ((1 << s) - 1);
    int op_tag = address >> (s + b);

    accessData(op_tag, op_s, verbose);
}
```
仅仅考虑'I'未考虑'M'
M = Load + Store, 必须访问两次
hit/miss/eviction都少算

   解决方案：
```c
    accessData(op_tag, op_s, verbose);

    if (operation == 'M')         
        accessData(op_tag, op_s, verbose);
```

3. 遇到问题：A[0][1]已经在cache里，那么B[1][0]不就不用访存了吗，为什么会有miss?
   解决方案：关键两点：
   1.A and B 使用不同的物理地址
   2.cache 按照地址索引不是按值索引
   A.tag != B.tag

4. 遇到问题：为什么对角线块会造成额外miss
   解决问题：
   核心原因：对角块A[i][i]和B[i][i]物理地址的set index相同
   针对A[i][0-7]这一行 and B[0-7][i]这一列
   对于1 line per set的直接映射cache这意味着readA and writeB来回evict,除去开头的cold miss 剩下全是conflict miss,没有hit

   为什么B[i][i]和A[i][i]拥有相同的set index?(以32x32为例)
   address_A[i][i] == i*32*4 + i*4
   address_B[i][i] == i*32*4 + i*4 + 32*32*4
   由s = 5; b = 5 => set_index == 0x1F & (address >> 5)
   set_index_A == 0x1F & (i*4)
   set_index_B == 0x1F & ((i*4) + (Binary)10000000) == 0x1F & (i*4) == set_index_A
   e.g
   read A[i][0]	    set S 为空         cold miss
   write B[0][i]	Set S → 有 tag_A   conflict miss，驱逐 A 的行，载入 B 的列块（tag_B）
   read A[i][1]	    Set S → 有 tag_B   conflict miss，驱逐 B，重新载入 A 的行
   write B[1][i]	Set S → 有 tag_A   conflict miss，再驱逐 A…
